version: '3.8'

services:
  # SPINE NODES (The Core of the Graph)
  spine:
    image: alpine:latest
    command: sh -c "apk add --no-cache iproute2 net-snmp && tail -f /dev/null"
    deploy:
      replicas: 4
    networks:
      - backbone_net

  # LEAF NODES (The Edge of the Graph)
  leaf:
    image: alpine:latest
    command: sh -c "apk add --no-cache iproute2 net-snmp openssh && tail -f /dev/null"
    deploy:
      replicas: 36
    networks:
      - backbone_net
    depends_on:
      - spine

networks:
  backbone_net:
    driver: bridge
    ipam:
      config:
        - subnet: 10.10.0.0/16

resources:
  limits:
    cpus: '0.1'
    memory: 64M
